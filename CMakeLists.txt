if(SDL2_FOUND)
	message(STATUS "SDL2 found")

macro(add_sdl_test name linkto)
  if(SDL2_FOUND)
    add_executable(${name} sdl2test.cpp)

    target_link_libraries(${name} ${SDL2_LIBRARIES} ${OPENGL_LIBRARIES} ${linkto})
  endif()
endmacro()
